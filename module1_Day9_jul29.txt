1.Get all the countries from Asia continent /region using Filter function

var url="https://restcountries.eu/rest/v2/all"
      var xhr = new XMLHttpRequest();
        xhr.open("GET", url);
        xhr.onload = function () {
          var data = JSON.parse(this.response);
         myfun(data)
         // console.log(data);
        };
        xhr.send();
        function myfun(a)
        {
          var b=a.filter(e=>{
              return e.region == "Asia"
          })
          console.log(b)
        }
